HTML Validation & Best Practices

✅ Good HTML Practices

<!-- Proper nesting -->
<div>
    <p>This paragraph is properly nested.</p>
    <ul>
        <li>List item 1</li>
        <li>List item 2</li>
    </ul>
</div>

<!-- Proper attributes -->
<img src="image.jpg" alt="Descriptive text" width="300" height="200">
<a href="https://example.com" title="Visit Example">Link</a>

❌ Common HTML Mistakes

<!-- Improper nesting -->
<p><div>Don't put div inside p</div></p>

<!-- Missing alt attribute -->
<img src="image.jpg">

<!-- Unclosed tags -->
<p>Paragraph without closing tag
<div>Another element</div>
